package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        try {
            // Veritabanını oluştur ya da aç
            SQLiteDatabase database = this.openOrCreateDatabase("musicians", MODE_PRIVATE, null);

            // Tabloyu oluştur, eğer mevcut değilse
            database.execSQL("CREATE TABLE IF NOT EXISTS musicians(name VARCHAR, age INT)");

            // Veriyi ekle
            database.execSQL("INSERT INTO musicians (name, age) VALUES ('James', 50)");

            // Verileri sorgula
            Cursor cursor = database.rawQuery("SELECT * FROM musicians", null);

            // İndeksleri al
            int nameIx = cursor.getColumnIndex("name");
            int ageIx = cursor.getColumnIndex("age");

            // Cursor ile veriyi işleme
            while(cursor.moveToNext()) {
                // İndeksleri kullanarak verileri al
                String name = cursor.getString(nameIx);
                int age = cursor.getInt(ageIx);

                // Veriyi kullan (örneğin loglamak ya da kullanıcıya göstermek)
                System.out.println("Name: " + name);
                System.out.println("Age: " + age);
            }
            cursor.close();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

Java-SQL Uygulama (Java 1)

package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

    EditText kadi, sifre;
    Button giris;
    SQLiteDatabase database;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        kadi = findViewById(R.id.txtKadi);
        sifre = findViewById(R.id.txtSifre);
        giris = findViewById(R.id.btnGiris);

        // Veritabanını oluştur veya aç
        database = this.openOrCreateDatabase("giris", MODE_PRIVATE, null);
        database.execSQL("CREATE TABLE IF NOT EXISTS giris (id INTEGER PRIMARY KEY AUTOINCREMENT, name VARCHAR, password VARCHAR)");
        database.execSQL("INSERT INTO giris (name, password) VALUES ('hasan', '1234')");

        giris.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String kullaniciAdi = kadi.getText().toString();
                String sifreGirilen = sifre.getText().toString();

                // Kullanıcı sorgusu
                Cursor cursor = database.rawQuery(
                    "SELECT * FROM giris WHERE name = ? AND password = ?",
                    new String[]{kullaniciAdi, sifreGirilen}
                );

                if (cursor.getCount() > 0) {
                    Toast.makeText(MainActivity.this, "Giriş Başarılı", Toast.LENGTH_SHORT).show();
                    cursor.close();
                    Intent intent = new Intent(MainActivity.this, KarsilamaEkrani.class);
                    startActivity(intent);
                } else {
                    Toast.makeText(MainActivity.this, "Hatalı Giriş", Toast.LENGTH_SHORT).show();
                    cursor.close();
                }
            }
        });
    }
}

package com.enderbad.artbook;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.view.Menu;
import android.view.MenuInflater;
import android.view.MenuItem;

public class MainActivity2 extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main2);
    }

    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        MenuInflater menuInflater = getMenuInflater();
        menuInflater.inflate(R.menu.add_art, menu);
        return true;
    }

    @Override
    public boolean onOptionsItemSelected(@NonNull MenuItem item) {
        if (item.getItemId() == R.id.add_art_item) {
            Intent intent = new Intent(MainActivity2.this, Main2Activity.class);
            startActivity(intent);
            return true;
        }
        return super.onOptionsItemSelected(item);
    }
}
